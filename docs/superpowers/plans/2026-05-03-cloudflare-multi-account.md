# Cloudflare Multi-Account Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Make the Cloudflare integration account-aware end-to-end so tunnels and zones never end up in mismatched accounts (the HTTP 530 / error 1033 bug).

**Architecture:** One config per org, holding the API token + an `accounts: jsonb [{id,name}]` array. Each `server` row stores `tunnelAccountId` (the account its tunnel was created in). At provisioning time we auto-derive the account when unambiguous, else require an explicit picker. `syncDomain` pre-flights the zone-to-tunnel account match. The misnamed "Reconcile" action becomes "Push to Cloudflare" for honesty.

**Tech Stack:** Drizzle ORM (PostgreSQL), tRPC 11, Next.js Pages Router, react-hook-form + zod, shadcn/ui, Vitest, Biome.

**Reference spec:** `docs/superpowers/specs/2026-05-03-cloudflare-multi-account-design.md`

```typescript
import type { CloudflareAccount } from "../../db/schema/cloudflare-config";

export type PickResult =
	| { kind: "ok"; accountId: string }
	| { kind: "ambiguous" }
	| { kind: "error"; message: string };

interface PickInput {
	accounts: CloudflareAccount[];
	zoneAccountIds: string[];
	explicitAccountId: string | null;
}

export const pickTunnelAccount = (input: PickInput): PickResult => {
	if (input.accounts.length === 0) {
		return { kind: "error", message: "Cloudflare config has no accounts" };
	}

	if (input.explicitAccountId) {
		const known = input.accounts.some(
			(a) => a.id === input.explicitAccountId,
		);
		if (!known) {
			return {
				kind: "error",
				message: `Account ${input.explicitAccountId} is not on this Cloudflare token`,
			};
		}
		return { kind: "ok", accountId: input.explicitAccountId };
	}

	if (input.accounts.length === 1) {
		return { kind: "ok", accountId: input.accounts[0]!.id };
	}

	const unique = Array.from(new Set(input.zoneAccountIds));
	if (unique.length === 1) {
		return { kind: "ok", accountId: unique[0]! };
	}

	return { kind: "ambiguous" };
};
```

```typescript
export const verifyToken = async (
	token: string,
): Promise<VerifyTokenResult> => {
	try {
		const verify = await cfFetch<{
			id: string;
			status: string;
			expires_on: string | null;
		}>(token, "GET", "/user/tokens/verify");

		const accounts: { id: string; name: string }[] = [];
		try {
			let page = 1;
			while (true) {
				const r = await cfFetch<Array<{ id: string; name: string }>>(
					token,
					"GET",
					`/accounts?per_page=50&page=${page}`,
				);
				accounts.push(
					...r.result.map((a) => ({ id: a.id, name: a.name })),
				);
				const total = r.result_info?.total_pages ?? 1;
				if (page >= total) break;
				page += 1;
			}
		} catch {
			// Accounts list may fail with insufficient scopes; verify still succeeded
		}

		return {
			ok: verify.result.status === "active",
			accounts,
			scopes: [],
			expiresOn: verify.result.expires_on,
			status: verify.result.status,
		};
	} catch (err) {
		if (err instanceof CloudflareApiError) {
			return {
				ok: false,
				accounts: [],
				scopes: [],
				expiresOn: null,
				status: err.status === 401 ? "invalid" : "error",
			};
		}
		throw err;
	}
};
```

```typescript
            .leftJoin(applications, eq(applications.applicationId, domains.applicationId))
            .leftJoin(compose, eq(compose.composeId, domains.composeId))
            .innerJoin(
                cloudflareZones,
                eq(domains.cloudflareZoneId, cloudflareZones.cloudflareZoneId),
            )
            .where(
                and(
                    isNotNull(domains.cloudflareZoneId),
                    or(
                        eq(applications.serverId, input.serverId),
                        eq(compose.serverId, input.serverId),
                    ),
                ),
            );
        return rows
            .filter((r) => r.zoneAccountId !== srv.tunnelAccountId)
            .map((r) => ({
                domainId: r.domainId,
                host: r.host,
                zoneName: r.zoneName,
                zoneAccountId: r.zoneAccountId,
                tunnelAccountId: srv.tunnelAccountId!,
            }));
    }),
```
